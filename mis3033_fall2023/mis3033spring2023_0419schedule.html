    function AddChart()
    {
        const ctx = document.querySelector('#myChart');
        if (mychart != null)
        {
            mychart.destroy();
        }
        ca = ctx.getContext('2d');
        ca.clearRect(0, 0, ctx.width, ctx.height);
        // ctx.innerHTML="";
        mychart = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'],
                datasets: [{
                    label: '# of Votes',
                    data: [12, 19, 3, 5, 2, 3],
                    borderWidth: 1
                }]
            },
            options: {}
        });
    }
</script>
