<script>
    var mychart = null;
    document.querySelector("#addchart").addEventListener("click", (e) =>
    {
        AddChart();
    });
    function AddChart()
    {
        const ctx = document.querySelector('#myChart');
        if (mychart != null)
        {
            mychart.destroy();
        }
        ca = ctx.getContext('2d');
        ca.clearRect(0, 0, ctx.width, ctx.height);
        // ctx.innerHTML="";
        mychart = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'],
                datasets: [{
                    label: '# of Votes',
                    data: [12, 19, 3, 5, 2, 3],
                    borderWidth: 1
                }]
            },
            options: {}
        });
    }
</script>

<script>

    "use strict";
    document.querySelector("#moveChart").addEventListener("click",(e)=>{move();});
    document.querySelector("#moveMap").addEventListener("click",(e)=>{moveMap();});
    function move ()
    {
        anime( {
            targets: document.querySelector( "#chartdiv" ),
            translateX: 270
        } );
        anime( {
                    targets: document.querySelectorAll( "div" ),
                    translateX: {
                        value: 200,
                        duration: 2000,
                    },
                    rotate: {
                        value: 180,
                        duration: 3000,
                    },
                    scale: {
                        value: 2,
                        duration: 3000,
                    },
                    keyframes: [
                        { translateY: -40 },
                        { translateX: 250 },
                        { translateY: 40 },
                        { translateX: 0 },
                        { translateY: 0 }
                    ],
                    duration: 4000,
                    //loop: true,
                } );
    }

    function moveMap ()
    {
        anime( {
            targets: document.querySelector( "#mapdiv" ),
            translateX: 270
        } );
    }
</script>

<script>
    document.querySelector("#showmap").addEventListener("click",(e)=>{drawmap();});
    function drawmap()
    {
        let map = L.map( 'mapdiv', { attributionControl: false, } ).setView( [ 35.33443889141701, -97.07270547900498 ], 12 );
        L.tileLayer( 'https://tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 19,
            attribution: ''
        } ).addTo( map );

        let marker = L.marker( [ 35.33443889141701, -97.07270547900498 ] ).addTo( map );
        marker.bindPopup( "<b>This is a lake</b><br>I work very hard!" ).openPopup();
    }
</script>
