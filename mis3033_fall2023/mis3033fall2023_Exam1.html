// define database tables
namespace eftest
{
    public class Student
    {
        [Key]// syntax to define the primary in a database table
        [DatabaseGenerated(DatabaseGeneratedOption.Identity)]// Auto increment
        public int Id { get; set; }
        public string name { get; set; } = "";
        public int age { get; set; } = 0;
    }
}


// define database connect class
using Microsoft.EntityFrameworkCore;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace eftest
{
    public class StuDB:DbContext
    {
        public DbSet<Student> Students { get; set; }// This is going to be a table in the database.
                    // You can add multiple database tables

        protected override void OnConfiguring(DbContextOptionsBuilder options)
        {

            options.UseSqlite("Data Source=Students.db");// Data Source=path to the database file
        }

    }
}


                

// Save to Database
StuDB stuDB = new StuDB();
Student stu = new Student() { name = "xinglong Ju", Id = 123, age = 35 };
stuDB.Students.Add(stu);
stuDB.SaveChanges();


// Read from Database
StuDB stuDB = new StuDB();// create a database connection
Student[] stus = stuDB.Students.ToArray<Student>();

// delete
var r=stuDB.Students.Where(x => x.stuID == 1238).FirstOrDefault();
stuDB.Remove(r);
stuDB.SaveChanges();

// modify
var r=stuDB.Students.Where(x => x.stuID == 1238).FirstOrDefault();
r.weight = 200;
stuDB.SaveChanges();
