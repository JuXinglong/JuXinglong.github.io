    function move ()
    {
        anime( {
            targets: document.querySelector( "#chartdiv" ),
            translateX: 270
        } );
        anime( {
                    targets: document.querySelectorAll( "div" ),
                    translateX: {
                        value: 200,
                        duration: 2000,
                    },
                    rotate: {
                        value: 180,
                        duration: 3000,
                    },
                    scale: {
                        value: 2,
                        duration: 3000,
                    },
                    keyframes: [
                        { translateY: -40 },
                        { translateX: 250 },
                        { translateY: 40 },
                        { translateX: 0 },
                        { translateY: 0 }
                    ],
                    duration: 4000,
                    //loop: true,
                } );
    }

    function moveMap ()
    {
        anime( {
            targets: document.querySelector( "#mapdiv" ),
            translateX: 270
        } );
    }

    function drawmap()
    {
        let map = L.map( 'mapdiv', { attributionControl: false, } ).setView( [ 35.33443889141701, -97.07270547900498 ], 12 );
        L.tileLayer( 'https://tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 19,
            attribution: ''
        } ).addTo( map );

        let marker = L.marker( [ 35.33443889141701, -97.07270547900498 ] ).addTo( map );
        marker.bindPopup( "<b>This is a lake</b><br>I work very hard!" ).openPopup();
    }
</script>
